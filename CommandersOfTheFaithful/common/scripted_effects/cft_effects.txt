﻿# Commanders of the Faithful - Scripted Effects
# These effects handle the Asabiyyah cycle mechanics
# Updated for CK3 1.19 (Scribe)

cft_apply_era = {
	remove_character_modifier = cft_era_hardship_modifier
	remove_character_modifier = cft_era_golden_modifier
	remove_character_modifier = cft_era_corruption_modifier
	remove_character_modifier = cft_era_dust_modifier

	set_variable = { name = cft_era value = $era$ }
	set_variable = { name = cft_progress value = 0 }

	if = {
		limit = { var:cft_era = 1 }
		add_character_modifier = cft_era_hardship_modifier
	}
	else_if = {
		limit = { var:cft_era = 2 }
		add_character_modifier = cft_era_golden_modifier
	}
	else_if = {
		limit = { var:cft_era = 3 }
		add_character_modifier = cft_era_corruption_modifier
	}
	else_if = {
		limit = { var:cft_era = 4 }
		add_character_modifier = cft_era_dust_modifier
	}

	# Store the year this era started
	set_variable = { name = cft_era_start_year value = current_year }
}

cft_initialize_asabiyyah = {
	if = {
		limit = { has_variable = cft_initialized }
	}
	else = {
		set_variable = { name = cft_initialized value = yes }
		set_variable = { name = cft_version value = 1 }
		set_variable = { name = cft_start_era value = 1 }

		# Check for historical context to set starting era
		# Note: e_abbasid and e_fatimid must exist in your mod or base game
		if = {
			limit = {
				current_date >= 868.1.1
				current_date < 1066.1.1
				has_title = e_abbasid
			}
			set_variable = { name = cft_start_era value = 3 }
		}
		if = {
			limit = {
				current_date >= 1066.1.1
				current_date < 1067.1.1
			}
			if = {
				limit = { has_title = e_abbasid }
				set_variable = { name = cft_start_era value = 4 }
			}
			else_if = {
				limit = { has_title = e_fatimid }
				set_variable = { name = cft_start_era value = 2 }
			}
		}
		if = {
			limit = { current_date >= 1178.1.1 }
			if = {
				limit = { has_title = e_abbasid }
				set_variable = { name = cft_start_era value = 4 }
			}
			else_if = {
				limit = { has_title = e_fatimid }
				set_variable = { name = cft_start_era value = 1 }
			}
		}

		cft_apply_era = { era = var:cft_start_era }
		remove_variable = cft_start_era
	}
}

cft_inherit_asabiyyah = {
	if = {
		limit = { exists = scope:predecessor }
		scope:predecessor = {
			if = {
				limit = { has_variable = cft_era }
				root = {
					set_variable = { name = cft_era value = prev.var:cft_era }
					if = { limit = { prev.has_variable = cft_progress } set_variable = { name = cft_progress value = prev.var:cft_progress } }
					if = { limit = { prev.has_variable = cft_era_start_year } set_variable = { name = cft_era_start_year value = prev.var:cft_era_start_year } }
					set_variable = { name = cft_initialized value = yes }
					if = { limit = { prev.has_variable = cft_version } set_variable = { name = cft_version value = prev.var:cft_version } }
					cft_apply_era = { era = prev.var:cft_era }
				}
			}
		}
	}
}

cft_tick_asabiyyah = {
	set_variable = { name = cft_temp_years_check value = current_year }
	if = {
		limit = { has_variable = cft_era_start_year }
		change_variable = { name = cft_temp_years_check subtract = var:cft_era_start_year }
	}
	else = {
		cft_initialize_asabiyyah = yes
	}

	if = {
		limit = { var:cft_temp_years_check < 5 }
		remove_variable = cft_temp_years_check
	}
	else = {
		set_variable = { name = cft_delta value = 0 }

		# ERA 1: HARDSHIP - Building cohesion through struggle
		if = {
			limit = { var:cft_era = 1 }
			if = {
				limit = { is_at_war = no }
				change_variable = { name = cft_delta add = 20 }
			}
			if = {
				limit = { NOT = { cft_has_dangerous_faction = { threshold = 50 } } }
				change_variable = { name = cft_delta add = 15 }
			}
			if = {
				limit = { gold > 100 }
				change_variable = { name = cft_delta add = 10 }
			}
			if = {
				limit = { prestige > 500 }
				change_variable = { name = cft_delta add = 10 }
			}
			if = {
				limit = {
					exists = capital_province
					capital_province = { county_control >= 70 }
				}
				change_variable = { name = cft_delta add = 10 }
			}
			if = {
				limit = { OR = { stewardship >= 15 learning >= 15 } }
				change_variable = { name = cft_delta add = 10 }
			}
			if = {
				limit = { cft_is_losing_war = yes }
				change_variable = { name = cft_delta subtract = 20 }
			}
			if = {
				limit = { cft_has_dangerous_faction = { threshold = 75 } }
				change_variable = { name = cft_delta subtract = 20 }
			}
			if = {
				limit = { gold < 0 }
				change_variable = { name = cft_delta subtract = 10 }
			}
			if = {
				limit = { tyranny >= 20 }
				change_variable = { name = cft_delta subtract = 10 }
			}
			if = {
				limit = { var:cft_delta < 2 }
				set_variable = { name = cft_delta value = 2 }
			}
		}
		# ERA 2: GOLDEN AGE - Peak cohesion, but seeds of decline
		else_if = {
			limit = { var:cft_era = 2 }
			if = {
				limit = { cft_is_over_limit = yes }
				change_variable = { name = cft_delta add = 15 }
			}
			if = {
				limit = { tyranny >= 20 }
				change_variable = { name = cft_delta add = 15 }
			}
			if = {
				limit = { cft_has_dangerous_faction = { threshold = 50 } }
				change_variable = { name = cft_delta add = 15 }
			}
			if = {
				limit = { is_at_war = yes }
				change_variable = { name = cft_delta add = 10 }
			}
			set_variable = { name = cft_temp_dur value = current_year }
			change_variable = { name = cft_temp_dur subtract = var:cft_era_start_year }
			if = {
				limit = { var:cft_temp_dur >= 10 }
				change_variable = { name = cft_delta add = 5 }
			}
			remove_variable = cft_temp_dur
			if = {
				limit = { stewardship >= 18 }
				change_variable = { name = cft_delta subtract = 15 }
			}
			if = {
				limit = { learning >= 18 }
				change_variable = { name = cft_delta subtract = 10 }
			}
			if = {
				limit = {
					gold > 200
					NOT = { any_joining_faction_count = { count >= 1 } }
				}
				change_variable = { name = cft_delta subtract = 10 }
			}
			if = {
				limit = { var:cft_delta < 2 }
				set_variable = { name = cft_delta value = 2 }
			}
		}
		# ERA 3: CORRUPTION - Decline accelerates
		else_if = {
			limit = { var:cft_era = 3 }
			if = {
				limit = { cft_is_losing_war = yes }
				change_variable = { name = cft_delta add = 20 }
			}
			if = {
				limit = { cft_has_dangerous_faction = { threshold = 75 } }
				change_variable = { name = cft_delta add = 20 }
			}
			if = {
				limit = { gold < 0 }
				change_variable = { name = cft_delta add = 15 }
			}
			if = {
				limit = {
					exists = capital_province
					capital_province = { county_control <= 40 }
				}
				change_variable = { name = cft_delta add = 15 }
			}
			if = {
				limit = { tyranny >= 30 }
				change_variable = { name = cft_delta add = 10 }
			}
			if = {
				limit = {
					has_trait = craven
					OR = { has_trait = greedy has_trait = callous has_trait = cynical }
				}
				change_variable = { name = cft_delta add = 10 }
			}
			if = {
				limit = { OR = { martial >= 18 stewardship >= 18 } }
				change_variable = { name = cft_delta subtract = 15 }
			}
			if = {
				limit = { NOT = { cft_has_dangerous_faction = { threshold = 50 } } }
				change_variable = { name = cft_delta subtract = 10 }
			}
			if = {
				limit = { gold > 200 }
				change_variable = { name = cft_delta subtract = 10 }
			}
			if = {
				limit = { var:cft_delta < 2 }
				set_variable = { name = cft_delta value = 2 }
			}
		}
		# ERA 4: DUST - Collapse and potential rebirth
		else_if = {
			limit = { var:cft_era = 4 }
			if = {
				limit = { NOT = { cft_is_losing_war = yes } }
				change_variable = { name = cft_delta add = 20 }
			}
			if = {
				limit = { NOT = { cft_has_dangerous_faction = { threshold = 50 } } }
				change_variable = { name = cft_delta add = 15 }
			}
			if = {
				limit = { gold > 0 }
				change_variable = { name = cft_delta add = 15 }
			}
			if = {
				limit = {
					exists = capital_province
					capital_province = { county_control >= 60 }
				}
				change_variable = { name = cft_delta add = 10 }
			}
			if = {
				limit = { OR = { learning >= 15 stewardship >= 15 } }
				change_variable = { name = cft_delta add = 10 }
			}
			set_variable = { name = cft_temp_dur value = current_year }
			change_variable = { name = cft_temp_dur subtract = var:cft_era_start_year }
			if = {
				limit = { var:cft_temp_dur >= 10 }
				change_variable = { name = cft_delta add = 10 }
			}
			remove_variable = cft_temp_dur
			if = {
				limit = { cft_is_losing_war = yes }
				change_variable = { name = cft_delta subtract = 20 }
			}
			if = {
				limit = { cft_has_dangerous_faction = { threshold = 75 } }
				change_variable = { name = cft_delta subtract = 20 }
			}
			if = {
				limit = { gold < 0 }
				change_variable = { name = cft_delta subtract = 10 }
			}
			if = {
				limit = { var:cft_delta < 5 }
				set_variable = { name = cft_delta value = 5 }
			}
		}

		change_variable = { name = cft_progress add = var:cft_delta }
		if = {
			limit = { var:cft_progress < 0 }
			set_variable = { name = cft_progress value = 0 }
		}
		if = {
			limit = { var:cft_progress > 100 }
			set_variable = { name = cft_progress value = 100 }
		}

		# Era transition at 100% progress
		if = {
			limit = { var:cft_progress >= 100 }
			set_variable = { name = cft_next_era value = var:cft_era }
			change_variable = { name = cft_next_era add = 1 }
			if = {
				limit = { var:cft_next_era > 4 }
				set_variable = { name = cft_next_era value = 1 }
			}
			cft_apply_era = { era = var:cft_next_era }
			trigger_event = { id = cft.0002 days = { 1 3 } }
			remove_variable = cft_next_era
		}
		
		remove_variable = cft_temp_years_check
	}
}
