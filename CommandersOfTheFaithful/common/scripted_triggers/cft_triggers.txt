		AND = {
			is_independent = yes
			# TODO verify clan-gov trigger for your version; swap to the next line if this fails:
			government_has_flag = flag:clan_government
			# government_has_flag = clan_government
			# is_clan_government = yes

		# TODO verify title ids (likely k_abbasid / k_fatimid if these never match). A wrong id here only fails silently.
