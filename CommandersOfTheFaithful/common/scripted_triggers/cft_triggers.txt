# Commanders of the Faithful - Scripted Triggers
# These triggers check conditions for Asabiyyah eligibility and state

cft_is_asabiyyah_eligible = {
	OR = {
		AND = {
			is_independent_ruler = yes
			has_government = clan_government
		}
		# Check for major caliphate titles that may use Asabiyyah regardless of government
		has_title = e_abbasid
		has_title = e_fatimid
	}
}

cft_is_losing_war = {
	any_character_war = {
		is_attacker = no
		war_score < -50
	}
}

cft_has_dangerous_faction = {
	any_joining_faction = {
		faction_power >= $threshold$
	}
}

cft_is_over_limit = {
	OR = {
		domain_limit_exceeded = yes
		vassal_limit_exceeded = yes
	}
}
