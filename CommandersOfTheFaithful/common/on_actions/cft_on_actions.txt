﻿# NOTE: relies on on_action merging across mods.
# Updated for CK3 1.19 (Scribe)

on_game_start = {
	events = { cft_on_game_start_init }
}

cft_on_game_start_init = {
	hidden = yes
	immediate = {
		every_living_character = {
			limit = {
				is_ruler = yes
				cft_is_asabiyyah_eligible = yes
			}
			cft_initialize_asabiyyah = yes
			trigger_event = { id = cft.0001 days = { 1 10 } }
		}
		every_living_character = {
			limit = {
				is_ruler = yes
				NOT = { cft_is_asabiyyah_eligible = yes }
			}
			remove_character_modifier = cft_era_hardship_modifier
			remove_character_modifier = cft_era_golden_modifier
			remove_character_modifier = cft_era_corruption_modifier
			remove_character_modifier = cft_era_dust_modifier
			set_variable = { name = cft_initialized value = no }
		}
	}
}

on_succession = {
	events = { cft.0004 }
}
