namespace = cft

cft_yearly_tick = {
	hidden = yes
	trigger = { is_alive = yes is_ruler = yes }
	immediate = {
		if = {
			limit = { cft_is_asabiyyah_eligible = yes }
			if = {
				limit = { NOT = { has_variable = cft_initialized } }
				cft_initialize_asabiyyah = yes
			}
			cft_tick_asabiyyah = yes
			trigger_event = { id = cft_yearly_tick days = { 365 425 } }
		}
		else = {
			remove_character_modifier = cft_era_hardship_modifier
			remove_character_modifier = cft_era_golden_modifier
			remove_character_modifier = cft_era_corruption_modifier
			remove_character_modifier = cft_era_dust_modifier
			set_variable = { name = cft_initialized value = no }
		}
	}
}

cft_era_change_player = {
	title = cft_era_change_title
	desc = {
		triggered_desc = { trigger = { has_variable = { name = cft_era value = 1 } } desc = cft_era_change_desc_hardship }
		triggered_desc = { trigger = { has_variable = { name = cft_era value = 2 } } desc = cft_era_change_desc_golden }
		triggered_desc = { trigger = { has_variable = { name = cft_era value = 3 } } desc = cft_era_change_desc_corruption }
		triggered_desc = { trigger = { has_variable = { name = cft_era value = 4 } } desc = cft_era_change_desc_dust }
	}
	trigger = {
		is_ai = no
		has_game_rule = cft_player_notifications_on
	}
	option = { name = cft_era_change_option_ok }
}
